Appointment Picker Examples

Render custom template

Define your own template containing a custom JavaScript invocation (via button). Definition:

  • inner: iterated input (optionally wrapped in some list item); must contain {{time}} placeholder and an optional {{disabled}} placeholder
  • outer: renders the frame, contains wrapper, title and optional functions; must contain {{innerHtml}} placeholder 

var ap = new AppointmentPicker(document.getElementById('time'), {
	interval: 15,
	startTime: 10,
	endTime: 18,
	disabled: ['11:20', '13:30', '14:00', '15:30', '16:45', '17:00'],
	template: {
		inner: '<li class="appo-picker-list-item {{disabled}}"><input type="button" value="{{time}}" {{disabled}}></li>',
		outer: '<div class="your-wrapper"><button class="your-close" onclick="ap.close()">Close</button><span class="your-title">Custom Markup</span><ul class="appo-picker-list">{{innerHtml}}</ul><button class="your-set-time" onclick="ap.setTime(\'15:00\')">Set 15:00</button></div>'
	}
});
